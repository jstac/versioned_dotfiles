## Zsh

# Change shell
sudo -s
chsh -s /bin/zsh root
exit
chsh -s /bin/zsh 
# And now log out and back in.
# Set up config files
cd ~
mv .zshrc /tmp
ln -s gh_synced/versioned_dotfiles/zsh/.zshenv .
mkdir .config/zsh
cd .config/zsh
ln -s ~/gh_synced/versioned_dotfiles/zsh/.zshrc .
# Set up themes and plugins, managed manually
mkdir plugins
mkdir themes
cd themes
git clone https://github.com/spaceship-prompt/spaceship-prompt
cd ../plugins
git clone https://github.com/zdharma-continuum/fast-syntax-highlighting
git clone https://github.com/jeffreytse/zsh-vi-mode
cd ~
source ~/gh_synced/versioned_dotfiles/zsh/.zshrc
